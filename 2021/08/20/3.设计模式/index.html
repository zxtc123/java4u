йқҷжҖҒвјҜвјҡвҪ…жі•жЁЎејҸпјҡе°ҶдёҠвҫҜзҡ„еӨҡдёӘвјҜвјҡвҪ…жі•зҪ®дёәйқҷжҖҒзҡ„пјҢдёҚйңҖиҰҒеҲӣе»әвјҜвјҡе®һдҫӢпјҢзӣҙжҺҘи°ғвҪӨеҚіеҸҜпјӣ

йҖӮвҪӨеңәжҷҜпјҡеҮЎжҳҜеҮәзҺ°дәҶвјӨйҮҸдёҚеҗҢз§Қзұ»зҡ„дә§е“ҒйңҖиҰҒеҲӣе»әпјҢе№¶дё”е…·жңүе…ұеҗҢзҡ„жҺҘвјқж—¶пјҢеҸҜд»ҘйҖҡиҝҮвјҜвјҡвҪ…жі•жЁЎејҸиҝӣвҫҸеҲӣе»әгҖӮвјӨеӨҡж•°жғ…еҶөдёӢпјҢжҲ‘д»¬дјҡйҖүвҪӨвҖ”вҖ”йқҷжҖҒвјҜвјҡвҪ…жі•жЁЎејҸгҖӮ

2.жҠҪиұЎе·ҘеҺӮжЁЎејҸ

еҲӣе»әеӨҡдёӘвјҜвјҡзұ»пјҢпјҲз”ҹдә§дёҖдёӘзі»еҲ—е•Ҷе“ҒпјүжҸҗвҫјвјҜвјҡзҡ„жү©еұ•жҖ§пјҢдёҚвҪӨеғҸдёҠвҫҜвјҖж ·еҰӮжһңеўһеҠ дә§е“ҒеҲҷиҰҒеҺ»дҝ®ж”№е”ҜвјҖзҡ„вјҜвјҡзұ»пјӣ

3.еҚ•дҫӢжЁЎејҸ

дҝқиҜҒеңЁвјҖдёӘJVMдёӯпјҢиҜҘеҜ№иұЎеҸӘжңүвјҖдёӘе®һдҫӢеӯҳеңЁпјӣ

public class Singleton {
    /* жҢҒжңүз§ҒжңүйқҷжҖҒе®һдҫӢпјҢйҳІвҪҢиў«еј•вҪӨпјҢжӯӨеӨ„иөӢеҖјдёәnullпјҢвҪ¬зҡ„жҳҜе®һзҺ°е»¶иҝҹеҠ иҪҪ */
    private state Singleton instance = null;
    /* з§Ғжңүжһ„йҖ вҪ…жі•пјҢйҳІвҪҢиў«е®һдҫӢеҢ– */
    private Singleton(){}
    /* йқҷжҖҒвјҜзЁӢвҪ…жі•пјҢеҲӣе»әе®һдҫӢ */
    public static Singleton getInstance(){
        if(instance == null){
            instance = new Singleton();
        }
        return instance;
    }
}

1гҖҒйҘҝжұүејҸпјҡзұ»еҲқе§ӢеҢ–ж—¶еҲӣе»әеҚ•дҫӢпјҢзәҝзЁӢе®үе…ЁпјҢйҖӮвҪӨдәҺеҚ•дҫӢеҚ еҶ…еӯҳвј©зҡ„еңәжҷҜпјҢеҗҰеҲҷжҺЁиҚҗдҪҝвҪӨжҮ’жұүејҸе»¶иҝҹеҠ иҪҪпјӣ

public class Singleton {
    private static Singleton instance = new Singleton();
    private Singleton(){}
    public static Singleton getInstance(){return instance;}
}

2гҖҒжҮ’жұүејҸпјҡйңҖиҰҒеҲӣе»әеҚ•дҫӢе®һдҫӢзҡ„ж—¶еҖҷеҶҚеҲӣе»әпјҢйңҖиҰҒиҖғиҷ‘зәҝзЁӢе®үе…Ё(жҖ§иғҪдёҚеӨӘеҘҪ)пјҡ

public class Singleton{
    private static Singleton instance = null;
    private Singleton(){}
    public static synchronized Singleton getInstance(){
        if(null == instance){
            instance = new Singleton();
        }
        return instance;
    }
}

3гҖҒеҸҢйҮҚжЈҖйӘҢй”Ғпјҡж•ҲзҺҮвҫјпјӣ(и§ЈеҶій—®йўҳпјҡеҒҮеҰӮдёӨдёӘзәҝзЁӢAгҖҒBпјҢAжү§вҫҸдәҶif (instance == null)иҜӯеҸҘпјҢе®ғдјҡи®ӨдёәеҚ•дҫӢеҜ№иұЎжІЎжңүеҲӣе»әпјҢжӯӨж—¶зәҝзЁӢеҲҮеҲ°Bд№ҹжү§вҫҸдәҶеҗҢж ·зҡ„иҜӯеҸҘпјҢBд№ҹи®ӨдёәеҚ•дҫӢеҜ№иұЎжІЎжңүеҲӣе»әпјҢз„¶еҗҺдёӨдёӘзәҝзЁӢдҫқж¬Ўжү§вҫҸеҗҢжӯҘд»Јз Ғеқ—пјҢе№¶еҲҶеҲ«еҲӣе»әдәҶвјҖдёӘеҚ•дҫӢеҜ№иұЎгҖӮ)

public class Singleton {
    private static volatile Singleton instance = null;//volatileзҡ„вјҖдёӘиҜӯд№үжҳҜзҰҒвҪҢжҢҮд»ӨйҮҚжҺ’еәҸдјҳеҢ–
    private Singleton(){}
    public static Singleton getInstance() {
        if(instance == null){
            synchronized(Singleton.class){
                if(instance == null){
                    instance = new Singleton();
                }
            }
        }
        return instance;
    }
}

4гҖҒйқҷжҖҒеҶ…йғЁзұ»вҪ…ејҸпјҡеҸҜд»ҘеҗҢж—¶дҝқиҜҒе»¶иҝҹеҠ иҪҪе’ҢзәҝзЁӢе®үе…ЁгҖӮ

public class Singleton {
    private static class SingletonHolder{
        public static Singleton instance = new Singleton();
    }
    private Singleton(){}
    public static Singleton getInstance(){
        return SingletonHolder.instance;
    }
}

4.еҺҹеһӢжЁЎејҸ

(еҜ№вјҖдёӘеҺҹеһӢеҜ№иұЎиҝӣвҫҸеӨҚеҲ¶гҖҒе…ӢйҡҶдә§вҪЈзұ»дјјж–°еҜ№иұЎ)пјҡе°ҶвјҖдёӘеҜ№иұЎдҪңдёәеҺҹеһӢпјҢеҜ№е…¶иҝӣвҫҸеӨҚеҲ¶гҖҒе…ӢйҡҶпјҢдә§вҪЈвјҖдёӘе’Ңе…ғеҜ№иұЎзұ»дјјзҡ„ж–°еҜ№иұЎпјӣ

1гҖҒж ёвјјпјҡе®ғзҡ„ж ёвјјжҳҜеҺҹеһӢзұ»PrototypeпјҢйңҖиҰҒе®һзҺ°CloneableжҺҘвјқпјҢе’ҢйҮҚеҶҷObjectзұ»дёӯзҡ„cloneвҪ…жі•пјӣ

2гҖҒдҪңвҪӨпјҡдҪҝвҪӨеҺҹеһӢжЁЎејҸеҲӣе»әеҜ№иұЎвҪҗзӣҙжҺҘnewвјҖдёӘеҜ№иұЎеңЁжҖ§иғҪдёҠиҰҒеҘҪзҡ„еӨҡпјҢеӣ дёәObjectзұ»зҡ„cloneвҪ…жі•жҳҜвјҖдёӘжң¬ең°вҪ…жі•пјҢе®ғзӣҙжҺҘж“ҚдҪңеҶ…еӯҳдёӯзҡ„вјҶиҝӣеҲ¶жөҒпјҢзү№еҲ«жҳҜеӨҚеҲ¶вјӨеҜ№иұЎж—¶пјҢжҖ§иғҪзҡ„е·®еҲ«вҫ®еёёжҳҺжҳҫгҖӮ

5.йҖӮй…ҚеҷЁжЁЎејҸ

е°ҶжҹҗдёӘзұ»зҡ„жҺҘвјқиҪ¬жҚўжҲҗе®ўжҲ·з«Ҝжңҹжңӣзҡ„еҸҰвјҖдёӘжҺҘвјқиЎЁзӨәпјҢвҪ¬зҡ„жҳҜж¶ҲйҷӨз”ұдәҺжҺҘвјқдёҚеҢ№й…ҚжүҖйҖ жҲҗзҡ„зұ»зҡ„е…је®№жҖ§й—®йўҳ

1гҖҒзұ»зҡ„йҖӮй…ҚеҷЁжЁЎејҸпјҡ

SourceжӢҘжңүдёҖдёӘж–№жі•пјҢзӣ®ж ҮжҺҘеҸЈTargetable,йҖҡиҝҮAdapterзұ»пјҢе°ҶSourceеҠҹиғҪжү©еұ•еҲ°Targetable

public class Source{
    public void method1(){
        System.out.println("this is original method!");
    }
}

public interface Targetable {
    //дёҺеҺҹзұ»дёӯж–№жі•зӣёеҗҢ
    public void method1();
    //ж–°зұ»зҡ„ж–№жі•
    public void method2();
}

public class Adapter extends Source implements Targetable {
    @Override
    public void method2(){
        System.out.println("this is the targetable method!");
    }
}

2гҖҒеҜ№иұЎзҡ„йҖӮй…ҚеҷЁжЁЎејҸпјҡ

public class Wrapper implements Targetable {
    private Source source;
    
    public Wrapper(Source source){
        super();
        this.source = source;
    }
    
    @Override
    public void method2(){
        System.out.println("this is the targetable method!");
    }
}

3гҖҒжҺҘвјқзҡ„йҖӮй…ҚеҷЁжЁЎејҸпјҡ

public interface Sourceable {
    public void method1();
    public void method2();
    public void method3();
}

public abstract class Wrapper2 implements Sourceable {
    public void method1(){}
    public void method2(){}
    public void method3(){}
}

public class SourceSub1 extends Wrapper2 {
    public void method1(){
        System.out.println("the sourceable interface's first Sub1!")
    }
}

public class SourceSub2 extends Wrapper2 {
    public void method2(){
        System.out.println("the sourceable interface's second Sub2!")
    }
}

4гҖҒдҪҝвҪӨеңәжҷҜпјҡ

1гҖҒзұ»зҡ„йҖӮй…ҚеҷЁжЁЎејҸпјҡеҪ“еёҢжңӣе°ҶвјҖдёӘзұ»иҪ¬жҚўжҲҗж»ЎвҫңеҸҰвјҖдёӘж–°жҺҘвјқзҡ„зұ»ж—¶пјҢеҸҜд»ҘдҪҝвҪӨзұ»зҡ„йҖӮй…ҚеҷЁжЁЎејҸпјҢеҲӣе»әвјҖдёӘж–°зұ»пјҢз»§жүҝеҺҹжңүзҡ„зұ»пјҢе®һзҺ°ж–°зҡ„жҺҘвјқеҚіеҸҜгҖӮ

2гҖҒеҜ№иұЎзҡ„йҖӮй…ҚеҷЁжЁЎејҸпјҡеҪ“еёҢжңӣе°ҶвјҖдёӘеҜ№иұЎиҪ¬жҚўжҲҗж»ЎвҫңеҸҰвјҖдёӘж–°жҺҘвјқзҡ„еҜ№иұЎж—¶пјҢеҸҜд»ҘеҲӣе»әвјҖдёӘWrapperзұ»пјҢжҢҒжңүеҺҹзұ»зҡ„вјҖдёӘе®һдҫӢпјҢеңЁWrapperзұ»зҡ„вҪ…жі•дёӯпјҢи°ғвҪӨе®һдҫӢзҡ„вҪ…жі•е°ұвҫҸгҖӮ

3гҖҒжҺҘвјқзҡ„йҖӮй…ҚеҷЁжЁЎејҸпјҡеҪ“дёҚеёҢжңӣе®һзҺ°вјҖдёӘжҺҘвјқдёӯжүҖжңүзҡ„вҪ…жі•ж—¶пјҢеҸҜд»ҘеҲӣе»әвјҖдёӘжҠҪиұЎзұ»WrapperпјҢе®һзҺ°жүҖжңүвҪ…жі•пјҢжҲ‘д»¬еҶҷеҲ«зҡ„зұ»зҡ„ж—¶еҖҷпјҢз»§жүҝжҠҪиұЎзұ»еҚіеҸҜгҖӮ
