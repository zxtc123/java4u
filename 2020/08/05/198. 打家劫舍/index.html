дҪ жҳҜдёҖдёӘдё“дёҡзҡ„е°ҸеҒ·пјҢи®ЎеҲ’еҒ·зӘғжІҝиЎ—зҡ„жҲҝеұӢгҖӮжҜҸй—ҙжҲҝеҶ…йғҪи—ҸжңүдёҖе®ҡзҡ„зҺ°йҮ‘пјҢеҪұе“ҚдҪ еҒ·зӘғзҡ„е”ҜдёҖеҲ¶зәҰеӣ зҙ е°ұжҳҜзӣёйӮ»зҡ„жҲҝеұӢиЈ…жңүзӣёдә’иҝһйҖҡзҡ„йҳІзӣ—зі»з»ҹпјҢеҰӮжһңдёӨй—ҙзӣёйӮ»зҡ„жҲҝеұӢеңЁеҗҢдёҖжҷҡдёҠиў«е°ҸеҒ·й—Ҝе…ҘпјҢзі»з»ҹдјҡиҮӘеҠЁжҠҘиӯҰгҖӮ

з»ҷе®ҡдёҖдёӘд»ЈиЎЁжҜҸдёӘжҲҝеұӢеӯҳж”ҫйҮ‘йўқзҡ„йқһиҙҹж•ҙж•°ж•°з»„пјҢи®Ўз®—дҪ  дёҚи§ҰеҠЁиӯҰжҠҘиЈ…зҪ®зҡ„жғ…еҶөдёӢ пјҢдёҖеӨңд№ӢеҶ…иғҪеӨҹеҒ·зӘғеҲ°зҡ„жңҖй«ҳйҮ‘йўқгҖӮ

зӨәдҫӢ 1пјҡ

иҫ“е…Ҙпјҡ[1,2,3,1]
иҫ“еҮәпјҡ4
и§ЈйҮҠпјҡеҒ·зӘғ 1 еҸ·жҲҝеұӢ (йҮ‘йўқ = 1) пјҢз„¶еҗҺеҒ·зӘғ 3 еҸ·жҲҝеұӢ (йҮ‘йўқ = 3)гҖӮ
еҒ·зӘғеҲ°зҡ„жңҖй«ҳйҮ‘йўқ = 1 + 3 = 4 гҖӮ

зӨәдҫӢ 2пјҡ

иҫ“е…Ҙпјҡ[2,7,9,3,1]
иҫ“еҮәпјҡ12
и§ЈйҮҠпјҡеҒ·зӘғ 1 еҸ·жҲҝеұӢ (йҮ‘йўқ = 2), еҒ·зӘғ 3 еҸ·жҲҝеұӢ (йҮ‘йўқ = 9)пјҢжҺҘзқҖеҒ·зӘғ 5 еҸ·жҲҝеұӢ (йҮ‘йўқ = 1)гҖӮ
еҒ·зӘғеҲ°зҡ„жңҖй«ҳйҮ‘йўқ = 2 + 9 + 1 = 12 гҖӮ

//з”ұдәҺдёҚеҸҜд»ҘеңЁзӣёйӮ»зҡ„жҲҝеұӢй—Ҝе…ҘпјҢжүҖд»Ҙ иҮіiжҲҝеұӢеҸҜзӣ—зӘғзҡ„жңҖеӨ§еҖјпјҢиҰҒд№Ҳе°ұжҳҜиҮі i-1 жҲҝеұӢеҸҜзӣ—зӘғзҡ„жңҖеӨ§еҖјпјҢ
//иҰҒд№Ҳе°ұжҳҜиҮі i-2 жҲҝеұӢеҸҜзӣ—зӘғзҡ„жңҖеӨ§еҖјеҠ дёҠеҪ“еүҚжҲҝеұӢзҡ„еҖјпјҢдәҢиҖ…д№Ӣй—ҙеҸ–жңҖеӨ§еҖјпјҢеҚіпјҡ
//dp[i] = max(dp[i-2]+nums[i], dp[i-1])
public static int rob(int[] nums) {
    int len = nums.length;
    if (len == 0) return 0;
    if (len == 1) return nums[0];
    int[] dp = new int[len];
    dp[0] = nums[0];
    dp[1] = Math.max(nums[0], nums[1]);

    for (int i = 2; i < len; i++) {
        dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i]);
    }

    return dp[len - 1];
}

/**
 * ж»ҡеҠЁж•°з»„
 * еҪ“еүҚжңҖеӨ§йҮ‘йўқдёҺеүҚ2й—ҙжҲҝзҡ„жңҖеӨ§йҮ‘йўқжңүе…іпјҢеҸӘи®°еҪ•еүҚ2й—ҙжҲҝйҮ‘йўқ
 *
 * @param nums
 * @return
 */
public int rob2(int[] nums) {
    if (nums == null || nums.length == 0) {
        return 0;
    }
    int length = nums.length;
    if (length == 1) {
        return nums[0];
    }
    int first = nums[0], second = Math.max(nums[0], nums[1]);
    for (int i = 2; i < length; i++) {
        int temp = second;
        second = Math.max(first + nums[i], second);
        first = temp;
    }
    return second;
}
